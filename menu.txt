# Menu driven program for the supply blockchain
while(1):
	print('\nChoose your option: ')
	print('-' * 30)
	print('1. Enter your transaction. ')
	print('2. Generate your RSA keys. ')
	print('3. Mine your block. ')
	print('4. View your blockchain. ')
	print('5. View your Unspent Transaction Output (UTXO). ')
	print('6. Verify the blockchain. ')
	print('7. Track an item.')
	print('8. Exit.')
	
	choice = int(input('Enter your option: '))
	
	if(choice == 1):
		make_transaction('','','')
	elif(choice == 2):
		number_manufacturers = int(input('\nEnter the number of manufacturers: '))
		generate_manufacturer_keys(number_manufacturers)
		number_other_users = int(input('Enter the number of stakeholders: '))
		generate_other_keys(number_other_users)
	elif(choice == 3):
		mine_block()
	elif(choice == 4):
		view_blockchain()
	elif(choice == 5):
		view_UTXO()
	elif(choice == 6):
		verify_blockchain()
	elif(choice == 7):
		item_code = input('Enter the item code: ')
		track_item(item_code)
	elif(choice == 8):
		break
	else:
		print('This is an invalid option.')
